# 请求拦截器

顾名思义 拦截器会在每个请求前执行该函数 可以在添加header中的验证token

	http.interceptors.request.use(
	  config => {
		  //config.data就是传输的数据 如果没有 设为空对象 
	    config.data = config.data ? config.data : {} 
	    config.headers['token'] = store.getters.token //在headers中添加token
	    return config
	  },
	  error => {
	    return Promise.error(error) //发生错误时抛出
	  })

# 调用方法二 优点：方便管理

	//封装一个请求函数 区分post 和 get
	function request(options) {
	  let tmpGet = {}
	  let tmpPost = {}
	  if (options.method === 'get') {
	    tmpGet = options.params
	  } else {
	    tmpPost = stringify(options.params)
	  }
	  return http.request({
	    url: options.url,
	    method: options.method,
	    data: tmpPost,
	    params: tmpGet
	  }).then(response => {
	    return Promise.resolve(response)
	  }, err => {
	    throw err
	  }).catch((error) => {
	    throw error
	  })
	}

	//创建一个api文件 统一管理所有api
	import request from '@/request/http.js'
	
	export const Login = (params) => {
	  return request({
	    url: '/shuiyuerp/user/login',
	    method: 'post',
	    params
	  })
	}
	
	//调用时导入进来
	login().then(res=>{})
